def dict_update(input_dict, update_dict):
    def _update(dict1, dict2):
        for key, value in dict2.iteritems():
            if key in dict1:
                if type(value) == dict:
                    _update(dict1[key], value)
                else:
                    dict1[key] = value
            else:
                dict1[key] = value
    _update(input_dict, update_dict)
